956 template <class el_type>
957 bool Adder<el_type>::add(uint8_t *idx, uint8_t idx_bits, el_type val, int proc_idx, uint8_t ini_flag) {
958  int *count = &send_cts_[proc_idx];
959  if (*count >= send_vals_.cols()) {
960  throw std::runtime_error("Too many elements added to Adder - must call perform_add() more frequently.");
961  }
962  uint8_t *cpy_idx = &send_idx_[proc_idx][*count * n_bytes_];
963  cpy_idx[n_bytes_ - 1] = 0; // To prevent buffer overflow in hash function after elements are added
964  memcpy(cpy_idx, idx, CEILING(idx_bits, 8));
965  if (ini_flag) {
966  set_bit(cpy_idx, idx_bits);
967  }
968  send_vals_(proc_idx, *count) = val;
969  (*count)++;
970  return (*count) < send_vals_.cols();
971 }

973 template <class el_type>
974 bool Adder<el_type>::add(uint8_t *idx, uint8_t idx_bits, el_type *vals, size_t n_vals, int proc_idx, uint8_t ini_flag) {
975  int *count = &send_cts_[proc_idx];
976  if (*count * n_vals >= send_vals_.cols()) {
977  throw std::runtime_error("Too many elements added to Adder in the process of taking a dot product - must call perform_dot() more frequently.");
978  }

980  cpy_idx[n_bytes_ - 1] = 0; // To prevent buffer overflow in hash function after elements are added

982  if (ini_flag) {
983  set_bit(cpy_idx, idx_bits);
984  }
985  std::copy(vals, vals + n_vals, &send_vals_(proc_idx, *count * n_vals));
986  (*count)++;
987  return (*count) < send_vals_.cols();
