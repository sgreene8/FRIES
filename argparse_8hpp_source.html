80  std::vector<std::string> inline split(const std::string &str) {
81  std::vector<std::string> splits;
82  std::stringstream ss(str);
83  std::string key;
84  while (std::getline(ss, key, ',')) {
85  if (!key.empty() && key.back() == '\0')
86  key.pop_back(); // last variables contain a '\0', which is unexpected when comparing to raw string, e.g. value == "test" will fail when the last character is '\0'. Therefore we can remove it
87  splits.emplace_back(std::move(key));
88  }
89  return splits;

110  template<typename T> inline T get(const std::string &v) { // remaining types
111  if constexpr (is_vector<T>::value) {
112  const std::vector<std::string> splitted = split(v);
113  T res(splitted.size());
114  if (!v.empty())
115  std::transform (splitted.begin(), splitted.end(), res.begin(), get<typename T::value_type>);
116  return res;
117  } else if constexpr (std::is_pointer<T>::value) {
118  return new typename std::remove_pointer<T>::type(get<typename std::remove_pointer<T>::type>(v));
119  } else if constexpr (is_shared_ptr<T>::value) {
120  return std::make_shared<typename T::element_type>(get<typename T::element_type>(v));
121  } else if constexpr (is_optional<T>::value) {
122  return get<typename T::value_type>(v);
123  } else if constexpr (std::is_enum<T>::value) { // case-insensitive enum conversion
124 #ifdef HAS_MAGIC_ENUM
125  constexpr auto& enum_entries = magic_enum::enum_entries<T>();
126  const std::string lower_str = to_lower(v);
127  for (const auto &[value, name] : enum_entries) {
128  if (to_lower(name) == lower_str)
129  return value;
130  }
131  std::string error = "enum is only accepting [";
132  for (int i = 0; i < enum_entries.size(); i++)
133  error += (i==0? "" : ", ") + to_lower(enum_entries[i].second);
134  error += "]";
135  throw std::runtime_error(error);
136 #else
137  throw std::runtime_error("Enum not supported, please install magic_enum (https://github.com/Neargye/magic_enum)");
138 #endif
139  } else {
140  return T(v);
141  }
142  }

356  /* parse all parameters and also check for the help_flag which was set in this constructor
357  * Upon error, it will print the error and exit immediately.
358  */
359  void parse(int argc, char *argv[]) {
360  program_name = argv[0];
361  params = std::vector<std::string>(argv + 1, argv + argc);
362 
363  bool& _help = flag("help", "print help");
364 
365  auto is_value = [&](const size_t &i) -> bool {
366  return params.size() > i && (params[i][0] != '-' || (params[i].size() > 1 && std::isdigit(params[i][1]))); // check for number to not accidentally mark negative numbers as non-parameter
367  };
368  auto parse_param = [&](size_t &i, const std::string &key, const bool is_short, const std::optional<std::string> &equal_value=std::nullopt) {
369  auto itt = kwarg_entries.find(key);
370  if (itt != kwarg_entries.end()) {
371  auto &entry = itt->second;
372  if (equal_value.has_value()) {
373  entry->_convert(equal_value.value());
374  } else if (entry->implicit_value_.has_value()) {
375  entry->_convert(*entry->implicit_value_);
376  } else if (!is_short) { // short values are not allowed to look ahead for the next parameter
377  if (is_value(i + 1)) {
378  std::string value = params[++i];
379  if (entry->_is_multi_argument) {
380  while (is_value(i + 1))
381  value += "," + params[++i];
382  }
383  entry->_convert(value);
384  } else if (entry->_is_multi_argument) {
385  entry->_convert(""); // for multiargument parameters, return an empty vector when not passing any more values
386  } else {
387  entry->error = "No value provided for: " + key;
388  }
389  } else {
390  entry->error = "No value provided for: " + key;
391  }
392  } else {
393  cerr << "unrecognised commandline argument: " << key << endl;
394  }
395  };
396  auto add_param = [&](size_t &i, const size_t &start) {
397  size_t eq_idx = params[i].find('='); // check if value was passed using the '=' sign
398  if (eq_idx != std::string::npos) { // key/value from = notation
399  std::string key = params[i].substr(start, eq_idx - start);
400  std::string value = params[i].substr(eq_idx + 1);
401  parse_param(i, key, false, value);
402  } else {
403  std::string key = std::string(params[i].substr(start));
404  parse_param(i, key, false);
405  }
406  };
407 
408  std::vector<std::string> arguments_flat;
409  for (size_t i = 0; i < params.size(); i++) {
410  if (!is_value(i)) {
411  if (params[i].size() > 1 && params[i][1] == '-') { // long --
412  add_param(i, 2);
413  } else { // short -
414  const size_t j_end = std::min(params[i].size(), params[i].find('=')) - 1;
415  for (size_t j = 1; j < j_end; j++) { // add possible other flags
416  const std::string key = std::string(1, params[i][j]);
417  parse_param(i, key, true);
418  }
419  add_param(i, j_end);
420  }
421  } else {
422  arguments_flat.emplace_back(params[i]);
423  }
424  }
425 
426  // Parse all the positional arguments, making sure multi_argument positional arguments are processed last to enable arguments afterwards
427  size_t arg_i = 0;
428  for (; arg_i < arg_entries.size() && !arg_entries[arg_i]->_is_multi_argument; arg_i++) { // iterate over positional arguments until a multi-argument is found
429  if (arg_i < arguments_flat.size())
430  arg_entries[arg_i]->_convert(arguments_flat[arg_i]);
431  }
432  size_t arg_j = 1;
433  for (size_t j_end = arg_entries.size() - arg_i; arg_j <= j_end; arg_j++) { // iterate from back to front, to ensure non-multi-arguments in the front and back are given preference
434  size_t flat_idx = arguments_flat.size() - arg_j;
435  if (flat_idx < arguments_flat.size() && flat_idx >= arg_i) {
436  if (arg_entries[arg_entries.size() - arg_j]->_is_multi_argument) {
437  std::stringstream s; // Combine multiple arguments into 1 comma-separated string for parsing
438  copy(&arguments_flat[arg_i],&arguments_flat[flat_idx + 1], std::ostream_iterator<std::string>(s,","));
439  std::string value = s.str();
440  value.back() = '\0'; // remove trailing ','
441  arg_entries[arg_i]->_convert(value);
442  } else {
443  arg_entries[arg_entries.size() - arg_j]->_convert(arguments_flat[flat_idx]);
444  }
445  }
446  }
447 
448  // try to apply default values for arguments which have not been set
449  for (const auto &entry : all_entries) {
450  if (!entry->value_.has_value()) {
451  entry->_apply_default();
452  }
453  }
454 
455  if (_help) {
456  help();
457  exit(0);
458  }
459 
460  validate();
461  }
