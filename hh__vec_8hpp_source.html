139  void find_neighbors_1D(uint8_t *det, uint8_t *neighbors) {
140  size_t n_elec = DistVec<el_type>::occ_orbs_.cols();
141  size_t byte_idx = 0;
142  size_t n_bytes = CEILING(n_sites_ * 2, 8);
143  uint8_t neib_bits[n_bytes];
144 
145  uint8_t mask = det[0] >> 1;
146  for (byte_idx = 1; byte_idx < n_bytes; byte_idx++) {
147  mask |= (det[byte_idx] & 1) << 7;
148  neib_bits[byte_idx - 1] = det[byte_idx - 1] & ~mask;
149 
150  mask = det[byte_idx] >> 1;
151  }
152  neib_bits[n_bytes - 1] = det[n_bytes - 1] & ~mask;
153 
154  zero_bit(neib_bits, n_sites_ - 1);
155  zero_bit(neib_bits, 2 * n_sites_ - 1); // open boundary conditions
156  if (((2 * n_sites_ - 1) % 8) != 0) {
157  neib_bits[n_bytes - 1] &= (1 << ((2 * n_sites_ - 1) % 8)) - 1;
158  }
159 
160  neighbors[0] = find_bits(neib_bits, &neighbors[1], n_bytes);
161 
162  mask = ~det[0] << 1;
163  neib_bits[0] = det[0] & mask;
164  for (byte_idx = 1; byte_idx < n_bytes; byte_idx++) {
165  mask = ~det[byte_idx] << 1;
166  mask |= (~det[byte_idx - 1] >> 7) & 1;
167  neib_bits[byte_idx] = det[byte_idx] & mask;
168  }
169  zero_bit(neib_bits, n_sites_); // open boundary conditions
170  if (((2 * n_sites_) % 8) != 0) {
171  neib_bits[n_bytes - 1] &= (1 << ((2 * n_sites_) % 8)) - 1;
172  }
173 
174  neighbors[n_elec + 1] = find_bits(neib_bits, &neighbors[n_elec + 1 + 1], n_bytes);
175  }

185  void decode_phonons(uint8_t *det, uint8_t *numbers) {
186  uint8_t mask = (1 << ph_bits_) - 1;
187  size_t curr_bit;
188  size_t curr_byte;
189  for (uint8_t site_idx = 0; site_idx < n_sites_; site_idx++) {
190  curr_bit = 2 * n_sites_ + site_idx * ph_bits_;
191  curr_byte = curr_bit / 8;
192  numbers[site_idx] = (det[curr_byte] >> (curr_bit % 8)) & mask;
193  if ((curr_bit % 8) + ph_bits_ > 8) {
194  numbers[site_idx] += (det[curr_byte + 1] << (8 - (curr_bit % 8))) & mask;
195  }
196  }
197  }

207  int det_from_ph(uint8_t *orig, uint8_t *new_det, uint8_t site_idx, int change) {
208  uint8_t bit_idx = 2 * n_sites_ + site_idx * ph_bits_;
209  uint16_t det_segment = orig[bit_idx / 8];
210  uint8_t n_bytes = CEILING(DistVec<el_type>::n_bits_, 8);
211  if (bit_idx / 8 + 1 < n_bytes) {
212  det_segment |= orig[bit_idx / 8 + 1] << 8;
213  }
214  uint16_t mask = (1 << ((bit_idx % 8) + ph_bits_)) - (1 << (bit_idx % 8));
215  uint16_t ph_num = (det_segment & mask) >> (bit_idx % 8);
216  if (change == 1 && ph_num == ((1 << ph_bits_) - 1)) {
217  std::cerr << "Warning: maximum phonon number reached\n";
218  return 0;
219  }
220  if (change == -1 && ph_num == 0) {
221  return 0;
222  }
223  ph_num += change;
224  ph_num <<= (bit_idx % 8);
225  det_segment &= ~mask;
226  det_segment |= ph_num;
227  memcpy(new_det, orig, n_bytes);
228  new_det[bit_idx / 8] = det_segment & 255;
229  if (bit_idx / 8 + 1 < n_bytes) {
230  new_det[bit_idx / 8 + 1] = det_segment >> 8;
231  }
232  return 1;
233  }
